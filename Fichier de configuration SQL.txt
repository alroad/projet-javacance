CREATE TABLE ARTISTE(
CODE_ARTISTE INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) not null,  
NOM VARCHAR(80) NOT NULL,
constraint PID primary key (CODE_ARTISTE));


CREATE TABLE ALBUM(CODE_ALBUM INTEGER GENERATED BY DEFAULT AS IDENTITY (START WITH 1, INCREMENT BY 1) not null PRIMARY KEY,
NOM VARCHAR(80) NOT NULL,
CODE_ARTISTE INTEGER);


CREATE TABLE CHANSON(ID INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1) NOT NULL PRIMARY KEY, NUMERO_CHANSON INTEGER NOT NULL, TITRE VARCHAR(80) NOT NULL, DUREE INTEGER NOT NULL,CODE_ALBUM INTEGER NOT NULL);
